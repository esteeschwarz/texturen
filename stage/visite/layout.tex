% quarto-page-geometry.tex
% Include this in the QUARTO SOURCE document (the one with your actual
% content), via header-includes, so that each logical page comes out
% at exactly 200mm x 240mm. The imposition step (impose.tex) relies on
% this exact size to avoid any rescaling.
%
% Usage in your .qmd YAML:
%
% format:
%   pdf:
%     documentclass: scrartcl

% ===================================================================
% LINE SPACING
% ===================================================================
\usepackage{setspace}

% With a scaled main font, set the font size and baseline skip together.
% This avoids the overly large spacing caused by mixing \setstretch and \linespread.
\AtBeginDocument{%
  \fontsize{11pt}{10.5pt}\selectfont
}
\setstretch{1.0}            % 1.0=single, 1.5=onehalfspacing, 2=double
% or use \singlespacing / \onehalfspacing / \doublespacing as commands
% to switch locally: \begin{spacing}{1.3} ... \end{spacing}

% ===================================================================
% HYPHENATION & JUSTIFICATION
% ===================================================================
% \usepackage[ngerman]{babel}    % or [english], drives hyphenation rules
%                                  % multiple: \usepackage[english,ngerman]{babel}

% \hyphenpenalty=300              % lower = more willing to hyphenate (default ~50)
% \tolerance=1000                 % how much bad spacing LaTeX accepts before hyphenating
% \emergencystretch=3em           % lets lines stretch instead of overfull boxes
% \sloppy                         % looser justification, fewer overfull hboxes (or \fussy to revert)

% \righthyphenmin=3               % min chars after a hyphen
% \lefthyphenmin=3                % min chars before a hyphen
% \hyphenation{Bei-spiel-wort}    % manual exceptions, space-separated list

% Disable hyphenation entirely for a block:
% \begin{sloppypar}\hyphenpenalty=10000 ... \end{sloppypar}

% ===================================================================
% PARAGRAPH SPACING / INDENT
% ===================================================================
\usepackage{parskip}            % no-indent, space-between-paragraphs style
% or manually:
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.6em plus 0.2em minus 0.1em}

% ===================================================================
% PAGE LAYOUT PARAMETERS (geometry package — fine-grained control)
% ===================================================================
\usepackage{geometry}
\geometry{
  paperwidth=120mm,
  paperheight=200mm,
  top=10mm,
  bottom=6mm,        % leave room for footer
  left=20mm,
  right=20mm,
%   headheight=12pt,     % must be >= header content height or you get a warning
%   headsep=8mm,         % gap between header and body text
  footskip=7mm,       % distance from body bottom to footer baseline
%   includehead,         % count header within the margin box (vs. extra space)
  includefoot          % same for footer
}

% ===================================================================
% HEADER / FOOTER CONTENT
% ===================================================================
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\fancyfootoffset{0pt}
\fancypagestyle{plain}{%
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}
\fancypagestyle{empty}{%
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}
\AtBeginDocument{\thispagestyle{empty}}

% ===================================================================
% MICROTYPE (subtle justification/kerning polish, big visual upgrade)
% ===================================================================
\usepackage{microtype}          % protrudes punctuation, tweaks letter spacing
                                  % just loading it improves justification quality

% Break before each top-level heading (H1 in the generated document).
% For scrartcl, this corresponds to \section.
\usepackage{etoolbox}
\pretocmd{\section}{\clearpage}{}{}

% ===================================================================
% WIDOW / ORPHAN CONTROL
% ===================================================================
\widowpenalty=10000              % discourage single line at top of page
\clubpenalty=10000                % discourage single line at bottom of page
